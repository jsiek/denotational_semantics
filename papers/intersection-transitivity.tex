\begin{align*}
  \dom{A → B} &= A \\
  \dom{A ∩ B} &= \dom{A} ∩ \dom {B} \\
  \\
  \cod{A → B} &= B \\
  \cod{A ∩ B} &= \cod{A} ∩ \cod {B}
\end{align*}
